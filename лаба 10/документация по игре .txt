"""
–ò–≥—Ä–∞ "–ö–∞–º–µ–Ω—å-–ù–æ–∂–Ω–∏—Ü—ã-–ë—É–º–∞–≥–∞" (üî• –°–£–õ–ò–§–ê-–±–∞—Ç–ª!)

–û–ø–∏—Å–∞–Ω–∏–µ:
–ò–≥—Ä–æ–∫ —Å–æ—Ä–µ–≤–Ω—É–µ—Ç—Å—è —Å –∫–æ–º–ø—å—é—Ç–µ—Ä–æ–º. –ù—É–∂–Ω–æ –≤—ã–±—Ä–∞—Ç—å –∫–∞–º–µ–Ω—å, –Ω–æ–∂–Ω–∏—Ü—ã –∏–ª–∏ –±—É–º–∞–≥—É.
–ö–æ–º–ø—å—é—Ç–µ—Ä –¥–µ–ª–∞–µ—Ç —Å–ª—É—á–∞–π–Ω—ã–π –≤—ã–±–æ—Ä. –ü–æ—Å–ª–µ –∫–∞–∂–¥–æ–≥–æ —Ä–∞—É–Ω–¥–∞ –≤—ã–≤–æ–¥–∏—Ç—Å—è —Ä–µ–∑—É–ª—å—Ç–∞—Ç.
–í –∫–æ–Ω—Ü–µ –æ–±—ä—è–≤–ª—è–µ—Ç—Å—è –ø–æ–±–µ–¥–∏—Ç–µ–ª—å –ø–æ —Å—É–º–º–µ –≤—Å–µ—Ö —Ä–∞—É–Ω–¥–æ–≤.

–ê–≤—Ç–æ—Ä: —Ç–≤–æ–µ –∏–º—è
"""

import random

# -----------------------------
# –í–∞—Ä–∏–∞–Ω—Ç—ã —Ö–æ–¥–æ–≤
# -----------------------------
ROCK = '–∫–∞–º–µ–Ω—å'
SCISSOR = '–Ω–æ–∂–Ω–∏—Ü—ã'
PAPER = '–±—É–º–∞–≥–∞'

# –°–ø–∏—Å–æ–∫ –≤—Å–µ—Ö —Ö–æ–¥–æ–≤
MOVES = [ROCK, SCISSOR, PAPER]

# -----------------------------
# –°—á–µ—Ç—á–∏–∫–∏ —Ä–µ–∑—É–ª—å—Ç–∞—Ç–æ–≤
# -----------------------------
player_win = 0  # –ø–æ–±–µ–¥—ã –∏–≥—Ä–æ–∫–∞
pc_win = 0      # –ø–æ–±–µ–¥—ã –∫–æ–º–ø—å—é—Ç–µ—Ä–∞
draw = 0        # –Ω–∏—á—å–∏

# -----------------------------
# –û—Å–Ω–æ–≤–Ω–æ–π –∏–≥—Ä–æ–≤–æ–π —Ü–∏–∫–ª
# -----------------------------

def play_round(player_choice, pc_choice):
    """
    –û–ø—Ä–µ–¥–µ–ª—è–µ—Ç —Ä–µ–∑—É–ª—å—Ç–∞—Ç –æ–¥–Ω–æ–≥–æ —Ä–∞—É–Ω–¥–∞ –∏–≥—Ä—ã.

    –ê—Ä–≥—É–º–µ–Ω—Ç—ã:
    player_choice (str): —Ö–æ–¥ –∏–≥—Ä–æ–∫–∞
    pc_choice (str): —Ö–æ–¥ –∫–æ–º–ø—å—é—Ç–µ—Ä–∞

    –í–æ–∑–≤—Ä–∞—â–∞–µ—Ç:
    str: 'player' –µ—Å–ª–∏ –≤—ã–∏–≥—Ä–∞–ª –∏–≥—Ä–æ–∫,
         'pc' –µ—Å–ª–∏ –≤—ã–∏–≥—Ä–∞–ª –∫–æ–º–ø—å—é—Ç–µ—Ä,
         'draw' –µ—Å–ª–∏ –Ω–∏—á—å—è,
         'invalid' –µ—Å–ª–∏ –Ω–µ–∫–æ—Ä—Ä–µ–∫—Ç–Ω—ã–π –≤–≤–æ–¥
    """
    if player_choice not in MOVES:
        return 'invalid'

    if player_choice == pc_choice:
        return 'draw'
    
    if (player_choice == ROCK and pc_choice == SCISSOR) or \
       (player_choice == SCISSOR and pc_choice == PAPER) or \
       (player_choice == PAPER and pc_choice == ROCK):
        return 'player'
    else:
        return 'pc'


def main():
    """
    –û—Å–Ω–æ–≤–Ω–∞—è —Ñ—É–Ω–∫—Ü–∏—è –∏–≥—Ä—ã. –ó–∞–ø—Ä–∞—à–∏–≤–∞–µ—Ç –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ —Ä–∞—É–Ω–¥–æ–≤,
    –ø—Ä–æ–≤–æ–¥–∏—Ç –∫–∞–∂–¥—ã–π —Ä–∞—É–Ω–¥, –≤—ã–≤–æ–¥–∏—Ç —Ä–µ–∑—É–ª—å—Ç–∞—Ç, –∞ –≤ –∫–æ–Ω—Ü–µ
    –æ–±—ä—è–≤–ª—è–µ—Ç –ø–æ–±–µ–¥–∏—Ç–µ–ª—è.
    """
    global player_win, pc_win, draw

    print("üî• –ô–æ, —ç—Ç–æ –°–£–õ–ò–§–ê-–±–∞—Ç–ª! üî•")
    rounds = int(input("–°–∫–æ–ª—å–∫–æ —Ä–∞—É–Ω–¥–æ–≤ –∑–∞–º—É—Ç–∏–º? "))

    for i in range(1, rounds + 1):
        print(f"\n–†–∞—É–Ω–¥ {i}/{rounds}")
        player = input('üëâ –¢–≤–æ–π —Ö–æ–¥ (–∫–∞–º–µ–Ω—å/–Ω–æ–∂–Ω–∏—Ü—ã/–±—É–º–∞–≥–∞): ').strip().lower()
        pc = random.choice(MOVES)
        print(f'ü§ñ –ö–æ–º–ø—å—é—Ç–µ—Ä –≤—ã–±—Ä–∞–ª: {pc}')

        result = play_round(player, pc)

        if result == 'draw':
            print('üòê –ù–∏—á—å—è‚Ä¶ —Ç–∞–∫ —Å–µ–±–µ –∑–∞–º–µ—Å.')
            draw += 1
        elif result == 'player':
            print('üî• –†–∞—É–Ω–¥ —Ç–≤–æ–π!')
            player_win += 1
        elif result == 'pc':
            print('üíÄ –ö–æ–º–ø—å—é—Ç–µ—Ä —Ç–µ–±—è —É–¥–µ–ª–∞–ª‚Ä¶ , —ç—Ç–æ —Ñ–∏–∞—Å–∫–æ –±—Ä–∞—Ç–∞–Ω!')
            pc_win += 1
        else:
            print('‚ö†Ô∏è –ù–µ–∫–æ—Ä—Ä–µ–∫—Ç–Ω—ã–π –≤–≤–æ–¥! –ü–æ–ø—Ä–æ–±—É–π —Å–Ω–æ–≤–∞.')

    # -----------------------------
    # –ò—Ç–æ–≥–æ–≤—ã–π —Ä–µ–∑—É–ª—å—Ç–∞—Ç
    # -----------------------------
    print("\nüèÅ –ò–≥—Ä–∞ –∑–∞–≤–µ—Ä—à–µ–Ω–∞!")
    print(f'üèÜ –ü–æ–±–µ–¥—ã –∏–≥—Ä–æ–∫–∞: {player_win}')
    print(f'ü§ñ –ü–æ–±–µ–¥—ã –∫–æ–º–ø—å—é—Ç–µ—Ä–∞: {pc_win}')
    print(f'üòê –ù–∏—á—å–∏: {draw}')

    if player_win > pc_win:
        print('üéâ –ü–æ–∑–¥—Ä–∞–≤–ª—è—é, —Ç—ã —á–µ–º–ø–∏–æ–Ω! –ú–∞—à–∏–Ω–∞ –≤ —à–æ–∫–µ –æ—Ç —Ç–≤–æ–µ–π –º–æ—â–∏.')
    elif player_win < pc_win:
        print('ü§ñ –ö–æ–º–ø—å—é—Ç–µ—Ä –ø–æ–±–µ–¥–∏–ª! –í–æ—Å—Å—Ç–∞–Ω–∏–µ –º–∞—à–∏–Ω —É–∂–µ –±–ª–∏–∑–∫–æ...')
    else:
        print('üòê –ù–∏—á—å—è. –ù—É, –ø–æ –∫—Ä–∞–π–Ω–µ–π –º–µ—Ä–µ, –Ω–∏–∫—Ç–æ –Ω–µ –ø–æ–∑–æ—Ä–∏–ª—Å—è.')


‚úÖ –ß—Ç–æ —Å–¥–µ–ª–∞–Ω–æ:

–í–µ—Å—å –ø—Ä–æ–µ–∫—Ç –≤ –æ–¥–Ω–æ–º —Ñ–∞–π–ª–µ.

–î–æ–±–∞–≤–ª–µ–Ω—ã —Ñ—É–Ω–∫—Ü–∏–∏ (play_round, main) –¥–ª—è —Å—Ç—Ä—É–∫—Ç—É—Ä–∏—Ä–æ–≤–∞–Ω–Ω–æ—Å—Ç–∏.

–ü–æ–ª–Ω—ã–µ docstring‚Äô–∏ –¥–ª—è –º–æ–¥—É–ª—è –∏ —Ñ—É–Ω–∫—Ü–∏–π.

–°—á–µ—Ç—á–∏–∫–∏ –∏ —Å–ø–∏—Å–æ–∫ —Ö–æ–¥–æ–≤ –≤—ã–Ω–µ—Å–µ–Ω—ã –≤ –∫–æ–Ω—Å—Ç–∞–Ω—Ç—ã.

–ü—Ä–æ–≤–µ—Ä–∫–∞ –Ω–µ–∫–æ—Ä—Ä–µ–∫—Ç–Ω–æ–≥–æ –≤–≤–æ–¥–∞.
